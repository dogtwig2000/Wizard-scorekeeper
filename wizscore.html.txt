<script>
    let players = [];
    let savedPlayers = [];
    const MAX_HANDS = 15;

    window.onload = function() {
        loadSavedPlayers();
        renderSavedPlayers();
    };

    function loadSavedPlayers() {
        const saved = localStorage.getItem('wizardSavedPlayers');
        if (saved) {
            savedPlayers = JSON.parse(saved);
        }
    }

    function saveSavedPlayers() {
        localStorage.setItem('wizardSavedPlayers', JSON.stringify(savedPlayers));
    }

    function savePlayer() {
        const input = document.getElementById('save-player-name');
        const name = input.value.trim();
        
        if (name) {
            if (!savedPlayers.includes(name)) {
                savedPlayers.push(name);
                saveSavedPlayers();
                renderSavedPlayers();
                input.value = '';
            } else {
                alert('This player is already saved!');
            }
        }
    }

    function removeSavedPlayer(name) {
        if (confirm(`Remove ${name} from saved players?`)) {
            savedPlayers = savedPlayers.filter(p => p !== name);
            saveSavedPlayers();
            renderSavedPlayers();
        }
    }

    function renderSavedPlayers() {
        const quickAddDiv = document.getElementById('quick-add-buttons');
        const savedListDiv = document.getElementById('saved-players-list');
        
        if (savedPlayers.length === 0) {
            quickAddDiv.innerHTML = '<div class="empty-state">No saved players yet</div>';
            savedListDiv.innerHTML = '';
            return;
        }
        
        quickAddDiv.innerHTML = savedPlayers.map(name => 
            `<button class="btn-quick" onclick="quickAddPlayer('${name}')">‚ú® ${name}</button>`
        ).join('');
        
        savedListDiv.innerHTML = '<div style="margin-top: 12px;">' + 
            savedPlayers.map(name => 
                `<span class="saved-player-tag">${name}<button class="btn-remove-tag" onclick="removeSavedPlayer('${name}')">‚úï</button></span>`
            ).join('') + '</div>';
    }

    function quickAddPlayer(name) {
        if (players.some(p => p.name === name)) {
            alert(`${name} is already in this game!`);
            return;
        }
        
        players.push({ name: name, scores: new Array(MAX_HANDS).fill(0), total: 0 });
        updatePlayerList();
        document.getElementById('start-btn').style.display = 'inline-block';
    }

    function addPlayer() {
        const input = document.getElementById('new-player-name');
        const name = input.value.trim();
        
        if (name) {
            if (players.some(p => p.name === name)) {
                alert(`${name} is already in this game!`);
                return;
            }
            
            players.push({ name: name, scores: new Array(MAX_HANDS).fill(0), total: 0 });
            input.value = '';
            updatePlayerList();
            document.getElementById('start-btn').style.display = 'inline-block';
            input.focus();
        }
    }

    function updatePlayerList() {
        const list = document.getElementById('player-list');
        if (players.length === 0) {
            list.textContent = "No players added yet";
        } else {
            list.innerHTML = `<strong>Players:</strong> ${players.map(p => p.name).join(', ')}`;
        }
    }

    function startGame() {
        if (players.length === 0) return alert("Add at least one player!");

        document.getElementById('setup-section').classList.add('hidden');
        document.getElementById('game-section').classList.remove('hidden');

        renderTable();
    }

    function renderTable() {
        const headerRow = document.getElementById('header-row');
        const footerRow = document.getElementById('footer-row');
        const tbody = document.getElementById('table-body');

        while (headerRow.children.length > 1) {
            headerRow.removeChild(headerRow.lastChild);
        }
        while (footerRow.children.length > 1) {
            footerRow.removeChild(footerRow.lastChild);
        }

        players.forEach((player, index) => {
            const th = document.createElement('th');
            th.textContent = player.name;
            headerRow.appendChild(th);

            const td = document.createElement('td');
            td.id = `total-${index}`;
            td.textContent = '0';
            footerRow.appendChild(td);
        });

        for (let hand = 1; hand <= MAX_HANDS; hand++) {
            const tr = document.createElement('tr');
            
            const tdNum = document.createElement('td');
            tdNum.className = 'hand-col';
            tdNum.textContent = hand;
            tr.appendChild(tdNum);

            players.forEach((player, pIndex) => {
                const td = document.createElement('td');
                
                const inputGroup = document.createElement('div');
                inputGroup.className = 'score-input-group';

                const inputsRow = document.createElement('div');
                inputsRow.className = 'score-inputs-row';

                const bidInput = document.createElement('input');
                bidInput.type = 'number';
                bidInput.placeholder = 'Bid';
                bidInput.className = 'bid-input';
                bidInput.dataset.player = pIndex;
                bidInput.dataset.hand = hand - 1;
                bidInput.dataset.type = 'bid';
                bidInput.min = '0';
                bidInput.max = hand;
                
                const trickInput = document.createElement('input');
                trickInput.type = 'number';
                trickInput.placeholder = 'Tricks';
                trickInput.className = 'trick-input';
                trickInput.dataset.player = pIndex;
                trickInput.dataset.hand = hand - 1;
                trickInput.dataset.type = 'tricks';
                trickInput.min = '0';
                trickInput.max = hand;

                bidInput.addEventListener('input', calculateAll);
                trickInput.addEventListener('input', calculateAll);

                inputsRow.appendChild(bidInput);
                inputsRow.appendChild(trickInput);

                const scoreDisplay = document.createElement('div');
                scoreDisplay.id = `score-${pIndex}-${hand-1}`;
                scoreDisplay.className = 'round-score';
                scoreDisplay.textContent = '-';

                inputGroup.appendChild(inputsRow);
                inputGroup.appendChild(scoreDisplay);
                td.appendChild(inputGroup);
                tr.appendChild(td);
            });

            tbody.appendChild(tr);
        }
    }

    function calculateAll() {
        players.forEach((player, pIndex) => {
            let runningTotal = 0;

            for (let handIdx = 0; handIdx < MAX_HANDS; handIdx++) {
                const bidInput = document.querySelector(`input[data-player="${pIndex}"][data-hand="${handIdx}"][data-type="bid"]`);
                const trickInput = document.querySelector(`input[data-player="${pIndex}"][data-hand="${handIdx}"][data-type="tricks"]`);
                const scoreDisplay = document.getElementById(`score-${pIndex}-${handIdx}`);

                const bid = bidInput.value;
                const tricks = trickInput.value;

                if (bid !== '' && tricks !== '') {
                    const bidNum = parseInt(bid);
                    const trickNum = parseInt(tricks);
                    let handScore = 0;

                    if (bidNum === trickNum) {
                        handScore = 20 + (10 * trickNum);
                    } else {
                        const diff = Math.abs(bidNum - trickNum);
                        handScore = -(10 * diff);
                    }

                    scoreDisplay.textContent = handScore > 0 ? `+${handScore}` : handScore;
                    scoreDisplay.className = 'round-score ' + (handScore >= 0 ? 'score-pos' : 'score-neg');
                    
                    runningTotal += handScore;
                } else {
                    scoreDisplay.textContent = '-';
                    scoreDisplay.className = 'round-score';
                }
            }

            const totalCell = document.getElementById(`total-${pIndex}`);
            totalCell.textContent = runningTotal;
            players[pIndex].total = runningTotal;
        });
        
        updateFloatingScoreboard();
    }

    function updateFloatingScoreboard() {
        const rankingsDiv = document.getElementById('score-rankings');
        if (!rankingsDiv) return;
        
        // Create a sorted copy of players
        const sortedPlayers = players.map((p, idx) => ({
            name: p.name,
            total: p.total,
            index: idx
        })).sort((a, b) => b.total - a.total);
        
        // Find the highest score
        const highestScore = sortedPlayers.length > 0 ? sortedPlayers[0].total : 0;
        
        rankingsDiv.innerHTML = sortedPlayers.map((player, rank) => {
            const isLeader = player.total === highestScore && player.total > 0;
            const scoreClass = player.total < 0 ? 'negative' : '';
            return `
                <div class="score-item ${isLeader ? 'leader' : ''}">
                    <span class="player-name">${player.name}</span>
                    <span class="player-score ${scoreClass}">${player.total}</span>
                </div>
            `;
        }).join('');
    }
